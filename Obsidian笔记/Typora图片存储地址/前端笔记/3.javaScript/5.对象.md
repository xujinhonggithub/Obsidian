##### 2.3.4.4.1 ==统计出现最多次数的案例==

```
var str = 'aadstadsaglmggnasg';
        var o = {};

        for (let i = 0; i < str.length; i++) {
            var chars =  str.charAt(i); //字符串的每个字符
            if (o[chars]) {             //o[chars] 得到的属性值
                o[chars]++;
            }else {
                o[chars] = 1;
            }
        }
        console.log(o);
        //遍历对象
        var max = 0;
        var ch = '';
        // k是属性名 o[k]是属性值
        for (var k in o){
            if (o[k] > max){
                max = o[k];
                ch = k;
            }
        }
        console.log(max);
        console.log('出现最多的字符 ' + ch );
```
